        def create
          cookies.delete :auth_token

          @ivar = MA[:user].new(params[MA[:single_resource]])
          set_ivar
          if @ivar.save
            self.current_ma_user = @ivar unless MA[:use_activation]
            redirect_back_or_default('/')

          else
            render :new
          end
        end
