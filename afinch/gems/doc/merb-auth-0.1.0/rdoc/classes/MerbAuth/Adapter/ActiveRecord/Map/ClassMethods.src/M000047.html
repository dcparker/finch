          def find_active_with_conditions(conditions)
            if MA[:user].instance_methods.include?("activated_at")
              MA[:user].with_scope(:find => {:conditions => "activated_at IS NOT NULL"}) do
                MA[:user].find(:first, :conditions => conditions)
              end
            else
              MA[:user].find(:first, :conditions => conditions)
            end
          end
